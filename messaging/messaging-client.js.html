'use strict';
let room = require('./message-room.js');

/**
 * Browser-side messaging sdk
 */
class MessagingClient {
  /**
   * Synchronously initiate a messaging client with an user access token
   * @constructor
   * @param  {string} userAccessToken an jwt token to identify an user
   */
  constructor(userAccessToken) {}

  /**
   * Connect to a chatroom, by chatroomid, which is defined by developers
   * Note you can promisify all the callback functions to transfer to promise style
   * @param  {string|object} room id string or room object
   * @param  {object} options
   * @param  {bool} options.autoCreate create a room automatically if it is not existed
   * @param  {object} options.events
   * @param  {function} options.events.message triggers when new message is arrived
   * @param  {function} options.events.user_add triggers when new user is arrived
   * @param  {function} options.events.user_remove triggers when user left the room
   * @param  {function} options.events.message_flush triggers when admin clean up all the messages
   * @param  {function} options.events.message_remove triggers when single message is removed
   * @param  {function} options.events.user_count triggers when server automatically braodcasting user count
   * @param  {function} options.events.chat_write_mode triggers when chat_write_mode is changed
   * @param  {function} options.events.message_input_interval triggers when message_input_interval is changed, usually
   * happend when message number is too large
   *
   */
  connect(room, options, callback) {}

  /**
   * Disconnect from specific room
   * @param  {string|object} room id string or room object
   * @param  {object}   options
   * @param  {Function} callback callback (error, result)
   */
  disconnect(room, options, callback) {}

  /**
   * [sendMessage description]
   * @param  {string|object} room id string or room object
   * @param  {string} message is message
   * @param  {object}   options
   * @param  {Function} callback callback(error, response)
   */
  sendMessage(room, message, options, callback) {}
  /**
   * update The Nick name for specific users
   * @param  {string|object} room id string or room object
   * @param  {string}   nickname update the nick name
   * @param  {object}   options
   * @param  {Function} callback callback(error)
   */
  updateNickName(room, nickname, options, callback) {}

  /**
   * Show all the connected room ids
   * @param {object} options
   * @param {integer} options.page the order of page default=1
   * @param {integer} options.perPage the number of element per page default=10
   * @return {array}  array of room objects
   */
  getConnectedRooms(options) {}

  /**
 * get all the Active users in a specific chat rooms
  * @param  {string}   room     room id
  * @param  {object}   options  [description]
  * @param {integer} options.page the order of page default=1
  * @param {integer} options.perPage the number of element per page default=10
  * @param  {getUsersCallback} callback callback(error, users)
  */
  getUsers(room, options, callback) {}
  /**
   * Callback function of getUsers
   * @callback getUsersCallback
   * @param error error object
   * @param users array of users
   */

  /**
 * get all the Active users in a specific chat rooms
  * @param  {string}   room     room id
  * @param  {object}   options  [description]
  * @param {integer} options.perPage the number of element per page default=10
  * @param  {Function} callback callback(error, messages)
  * @return  {array} messages array of active messages
  */
  getMessages(room, options, callback) {}
}

module.exports = MessagingClient;
